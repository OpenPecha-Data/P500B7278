༤༡༥༽ ཡང་ཟབ་དུམ་བུའི་སྐོར་ལས། རྨི་ལམ་རང་གྲོལ་གདམས་པ་གསལ་བའི་སྒྲོན་མེ་བཞུགས། ༤༡༦༽དཔལ་ལྡན་རྡོ་རྗེ་ཕག་མོ་ལ་ཕྱག་འཚལ་ལོ། གང་གི་རང་བཞིན་འོད་གསལ་འདུས་མ་བྱས། །རྣམ་ཐར་གསུམ་ལྡན་སྤྲོས་པའི་མཐའ་དང་བྲལ། །རྣམ་ཀུན་མཆོག་ལྡན་དག་པ་གཉིས་ཀྱི་བདག །དབྱིངས་ལ་གཤེགས་བཞུགས་བརྗོད་པ་མི་མངའ་ཡང༌། །འགགས་མེད་རང་གསལ་ཡེ་ཤེས་ཆོ་འཕྲུལ་ལས། །སྣ་ཚོགས་ཐབས་མང་གང་འདུལ་མཛད་པ་ཡི། །བགྲང་ཡས་ཆོས་ཁམས་རྒྱ་མཚོའི་སྤྲིན་དུ་ཤར། །ཡང་ཟབ་མན་ངག་རྨི་ཁྲིད་གདམས་པའི་སྐོར། །སྐལ་ལྡན་བུ་ཡི་དོན་དུ་འདི་ ༤༡༧༽རུ་བསྟན། དེ་ལ་འདིར་རྨི་ལམ་ཆོས་སྐུར་གྲོལ་བའི་གདམས་པ་ཉམས་སུ་ལེན་པ་ལ། གཞི་ཁྲེགས་ཆོད་ཀྱི་ལྟ་བ་ཇི་བཞིན་པར་ཉམས་སུ་བླངས་ནས་ལམ་སྣང་རྨི་ལམ་སྦྱོང་བ་ལ། དང་པོ་འབྱུང་བ་ལྔ་པོའི་ཐིམ་རིམ་ངོས་བཟུང་བ་དང༌། གཞིའི་འོད་གསལ་སྐྱོན་གསུམ་དང་བྲལ་བར་ངོ་སྤྲོད་པ། ཁམས་ཀྱི་བྱེ་བྲག་དབྱེ་ཞིང་གེགས་སེལ་བ། འོད་གསལ་ཕྱི་ནང་གི་ངོ་བོ་བསྟན་ཅིང་འབྱུང་བ་གོ་འཕོས་ནས་གཉིས་མེད་དུ་འགགས་བསྡམས་པའོ། དེ་ནས་དང་པོ་རྨི་ལམ་གཉིད་དུ་ཕྱོགས་པའི་ཚེ་ན་ཤ་ཁམས་ས་ལ་ཐིམ་པའི་རྟགས་ ༤༡༨༽སུ་ལུས་ལྕི་ཞིང་འགུལ་སྙིང་མི་འདོད་པ་དང༌། ཁྲག་ཁམས་ཆུ་ལ་ཐིམ་པའི་རྟགས་སུ་ཁ་དང་སྣ་ལ་སོགས་པའི་ཆུ་རང་འབྱུང་ཞིང་ཚགས་མི་ཐུབ་པ་དང༌། དྲོད་ཁམས་མེ་ལ་ཐིམ་པའི་རྟགས་སུ་ཚ་ནད་ཀྱིས་གདུང་བ་བཞིན་མགོ་མཇུག་མེད་པར་དྲོད་འབར་བ། དབུགས་ཁམས་རླུང་ལ་ཐིམ་པའི་རྟགས་སུ་རླུང་ཕྱོགས་གཅིག་ཏུ་འདུས་ཤིང་དབུགས་འབྱིན་འཇུག་གི་རྒྱུན་ཧ་ཅང་ཐུང་ལ་ཤུགས་དྲག་པ། སེམས་ཁམས་ནམ་མཁའ་ལ་ཐིམ་པའི་རྟགས་སུ་ཉིན་སྣང་གི་རྟོག་པའི་འཕྲོ་ཆད་ནས་རྨི་སྣང་གི་ཡུལ་མ་མཆེད་པའི་བར་དུ་འོད་གསལ་གྱི་ངོ་བོ་བཟོད་མེད་དུ་འཆར་བ་ཡང་འབྱུང་ངོ༌། དེ་ལ་ཉིན་སྣང་གི་རྟོག་པས་མ་བསླད་རྨི་སྣང་གི་ཡུལ་མ་འཛིན་ཉིང་འཁྲུལ་གྱི་རྐྱེན་དང་བྲལ་བའི་བར་དུ་གཞིའི་འོད་གསལ་བཟོད་མེད་དུ་ཤར་བ་དེ་ལ་ཡང་ཡང་གོམས་འདྲིས་སོང་བས་ཉིན་མཚན་འོད་གསལ་ཁོར་ཡུག་ཏུ་གྲོལ་བའི་གདིང་རྙེད་དོ།

དེའི་སྐབས་སུ་རང་བཞིན་པ་རྣམས་ལ་ཁམས་ཀྱི་བྱེ་བྲག་གིས་ས་ཁམས་པ་རྣམས་ལ་སྣང་བ་སེར་ཐོམ་ཐོམ་བྱེད་ཅིང་ཉམས་མ་ངེས་པ་དུ་མ་འབྱུང་ངོ༌། ཆུ་ཁམས་ས་ལ་སྣང་བ་སྐྱ་རིབ་རིབ་འཆར་ཞིང་གསལ་ ༤༡༩༽ཉམས་ལ་སོགས་པ་གཞན་རྒྱུད་ཤེས་པ་ཡང་ཕྲན་བུ་རེ་འབྱུང༌། མེ་ཁམས་པ་ལ་སྣང་བ་འོད་དམར་གྱི་ལེ་བ་ཤར་ཞིང་ཉམས་ཀྱང་རྩུབ་ཅིང་འགྱུར་ཆ་ཆེ་བའོ། རླུང་ཁམས་པ་ལ་སྣང་བ་འཚུབས་ཆེ་ཞིང་སྲིད་པ་འཇིག་པ་དང་རི་ཉིལ་བ་ལྟ་བུའི་ཉམས་སྣང་འབྱུང༌། ནམ་མཁའི་ཁམས་ལ་གཉིད་མཐུག་ཅིང་རྟོག་ཚོགས་བཟང་ངན་གང་ལའང་ཆ་སྙོམས་ཤིང་སྐབས་རེར་འོད་གསལ་དང་པོའི་སྣང་བ་འོད་ཀྱི་རྣམ་པར་སྔོ་ཀྱི་ལེ་ཤར་རོ། ལྷ་དང་བླ་མའི་ཞལ་གཟིགས་དངོས་མེད་འགྲོ་བའི་ཤེས་རྒྱུད་ཕྲན་བུ་རེ་མཁྱེན་པ་འབྱུང་ངོ༌། དེ་དག་གང་སྣང་ཐམས་ཅད་རོལ་པ་དང་འཕྲུལ་པ་ཡིན་པས་བདེན་ཞེན་དང་སྤང་གཉེན་མེད་པར་རང་གྲོལ་དུ་ངོ་ཤེས་པར་གྱིས་ཤིག ས་འབྱུང་བ་ཀུན་གྱི་སྙིགས་མ་ཡིན་པས་འགྱུ་ཆ་དཀའ་ཞིང་ལྡོག་སླ་བས་སྣང་བ་མི་མངོན་ཅིང་མ་ངེས་པའོ། ཆུ་འབྱུང་བ་ཀུན་གྱི་གསོས་ལྟ་བུ་ཡིན་པས་འགྱུ་ཆ་མྱུར་གསལ་དང་ལྡན་ཅིང་སྣང་བ་བཟང་ངན་བར་གསུམ་འགགས་མེད་དུ་འཆར་ཞིང་གསལ་བའོ། མེ་ ༤༢༠༽འབྱུང་བ་ཀུན་གྱི་གྲོགས་ལྟ་བུ་ཡིན་པས་ཕྲ་རགས་ཀྱི་རྟོག་པ་ཐམས་ཅད་མངོན་དུ་བྱུང་ཞིང་སྣང་བ་མི་འཛིན་པར་འོད་གསལ་འཆར་བའི་རྒྱུ་བྱེད་དོ། རླུང་འབྱུང་བ་ཀུན་གྱི་རྒྱུ་ཡིན་པས་འགྱུ་ཆ་ཤིན་ཏུ་འཕྲོ་སླ་ཞིང་མྱུར་ལ་ཚུགས་མི་ཐུབ་ཅིང་མ་ངེས་སོ། ནམ་མཁའ་འབྱུང་བ་ཀུན་གྱི་གཞི་ཡིན་པས་འགྱུ་ཆ་ཟིན་སླ་བ་ལྡོག་སླ་བ་ངལ་དུབ་མེད་པར་རང་གྲོལ་དུ་འགྲོ་བའོ། དེ་དག་ངོ་ཤེས་ཤིང་དོན་རྟོགས་པའི་རྨི་ལམ་གྱི་སྐབས་རྟོག་ཚོགས་བཟང་ངན་བར་གསུམ་ཐམས་ཅད་ལ་གསལ་སྒྲིབ་མེད་ཀྱང་ཡིད་དཔྱོད་དང་བྲལ་ཞིང་འོད་གསལ་ཆོས་ཉིད་ཀྱི་རང་བཞིན་དུ་གོ་འཕོས་ནས་གཉིས་འཛིན་ཞེན་རྟོག་ཐམས་ཅད་ཐད་ཀ་རང་གྲོལ་དུ་ཞིག་པ་དང༌།

མཐུག་པོའི་འོད་གསལ་མངོན་དུ་གྱུར་ནས་བདག་གཞན་གཉིས་སུ་མེད་ཅིང་དབྱིངས་སུ་ཟད་ནས་ཆགས་སྡང་འཁྲུལ་རྟོག་ཐམས་ཅད་ཡེ་གྲོལ་ཆེན་པོའི་ངང་དུ་རྒྱ་ཡན་དུ་གནས་པས། བདག་འཛིན་གྱི་རྩ་བ་སྟོང་རྟོག་གཉིས་ཀྱི་སེམས་ཟད། གཟུང་ཡུལ་འཛིན་པ་བྲལ་བའི་ཉིན་སྣང་དང་རྨི་སྣང་གཉིས་དག་མཉམ་དུ་གྲོལ། མ་རིག་པ་གཞི་ལ་མི་ལྡན་ཞིང་རིག་པ་རང་བྱུང་གི་སངས་རྒྱས་ཆོས་ཉིད་མི་ ༤༢༡༽འགྱུར་བའི་གདིང་ཚེ་འདིར་ཐོབ་པར་འགྱུར་རོ། འབྱུང་བ་དེ་དག་ཀྱང་ཐ་དད་དུ་ཡོད་པ་མ་ཡིན་ཏེ་ངོ་བོ་ཡེ་ཤེས་ལྔ་ལས་མ་འདས་ཤིང་རྣམ་པ་ཡེ་ནས་མཐའ་གྲོལ་གཉིས་སུ་མེད་པ་རང་བྱུང་ལྷུན་རྫོགས་ཆེན་པོ་ནི། ས་ཁམས་སེར་པོར་སྣང་བ་དེ་ཤེས་རབ་སྟོང་པ་ཡུམ་གྱི་རང་བཞིན། སྙིགས་མ་ས་རུ་སྣང་བ་དེ་ཐབས་བདེ་ཆེན་ཡབ་ཀྱི་རང་བཞིན། ཆུ་ཁམས་དཀར་པོར་སྣང་བ་དེ་ཤེས་རབ་སྟོང་པ་ཡུམ་གྱི་རང་བཞིན། སྙིགས་མ་ཆུ་རུ་སྣང་བ་དེ་ཐབས་བདེ་ཆེན་ཡབ་ཀྱི་རང་བཞིན། མེ་ཁམས་དམར་པོར་སྣང་བ་དེ་ཤེས་རབ་ཡུམ་གྱི་རང་བཞིན། སྙིགས་མ་མེ་རུ་སྣང་བ་དེ་ཐབས་བདེ་ཆེན་ཡབ་ཀྱི་རང་བཞིན། རླུང་ཁམས་ལྗང་གུར་སྣང་བ་དེ་ཤེས་རབ་སྟོང་པ་ཡུམ་གྱི་རང་བཞིན། སྙིགས་མ་རླུང་དུ་སྣང་བ་དེ་ཐབས་བདེ་ཆེན་ཡབ་ཀྱི་རང་བཞིན། ནམ་མཁའི་ཁམས་མཐིང་གར་སྣང་བ་དེ་ཤེས་རབ་སྟོང་པ་ཡུམ་གྱི་རང་བཞིན། ནམ་མཁའ་བེམ་སྟོང་དུ་སྣང་བ་དེ་ཐབས་བདེ་ཆེན་ཡབ་ཀྱི་རང་བཞིན། ཤེས་རབ་ཀྱི་རོལ་པ་དོན་དམ་པ། ཐབས་ཀྱི་རོལ་པ་ཀུན་རྫོབ་ཏུ་ཁ་དྲང་བ་ཡིན་ནོ། མ་རིག་པའི་དབང་གིས་འབྱུང་བ་ལ་བདེ་ཞེན་དང་དངོས་པོར་ ༤༢༢༽བལྟས་པས་འབྱུང་དང་འབྱུང་འགྱུར་དུ་སྣང་ཡང་དོན་ལ་ཁ་དོག་ལྔ་འབྱུང་ལྔ་ཡབ་ལྔ་ཡུམ་ལྔ་ཡེ་ཤེས་ལྔ་ཤེས་རབ་ལ་སོགས་པ་ཐམས་ཅད་ཡེ་ནས་གཉིས་མེད་ལྷ་རུ་ཤེས་པར་བྱ་བ་གནད་ཀྱི་གདམས་པ་ཡིན་པས་བརྟག་དཔྱད་གོམས་འདྲིས་ཀྱི་སྒོ་ནས་ཉམས་སུ་ལེན་པར་བྱའོ། དེ་ལྟར་གནད་དང་ལྡན་ན་རབ་འབྱུང་བ་ཐིམ་རིམ་ལ་མི་ལྟོས་ཤིང་ནམ་མཁའ་ནམ་མཁའ་ལ་ཐིམ་པ་ལྟར་འོད་གསལ་ཀ་དག་ཆོས་སྐུར་གྲོལ་བ་དང༌།

འབྲིང་གིས་འབྱུང་བའི་སྙིགས་མ་དྭངས་མ་རུ་གོ་འཕོས་ནས་རྨི་ལམ་ངོས་ཟིན་ཅིང་སྤྲུལ་བསྒྱུར་སོགས་ལ་དབང་ཞིང་ལོངས་སྐུར་གྲོལ་བ་དང༌། ཐ་མས་ཀྱང་འབྱུང་བ་རང་ངོ་འཕྲོད་ནས་བདེན་མེད་རང་སྣང་དུ་རྟོགས་ཞིང་རྨི་ལམ་ངན་གོམས་ཀྱི་རྒྱུན་ཆད་ཅིང་དག་པའི་སྣང་བ་ལྷ་དང་དཀྱིལ་འཁོར་ཞིང་ཁམས་སོགས་མཐོང་བར་བྱེད་ཅིང་སྤྲུལ་སྐུར་གྲོལ་བར་འགྱུར་རོ། ཡང་ཟབ་རྨི་ཁྲིད་ཀྱི་མན་ངག་རྫོགས་སོ། ས་མ་ཡ། རྒྱ་རྒྱ་རྒྱ། ཞེས་པ་འདི་ཡང་གཏེར་ཆེན་པདྨ་འགྲོ་འདུལ་གསང་སྔགས་གླིང་པའི་དངོས་སློབ་དམ་པར་གྱུར་པ་དགེ་བཙུན་ཆོས་གསལ་གྱིས་ཡང་ཡང་བསྐུལ་ངོ་མི་ལྡོག་ཙམ་དུ་སློབ་འབངས་ཀུན་གྱི་ཐ་ཤལ་བ་བདེ་བའི་རྡོ་རྗེས་ ༤༢༣༽གང་ཤར་རྫུན་རིས་སུ་བྲིས་པའོ།། དགེའོ།། ཧ་ཧ། ཤ་ཟིན་སྤུ་ཟིན་གཏེར་མིན་གཏེར་ཆོས་འདི། །ཇོ་མིན་ནག་མིན་བདེ་བའི་མིང་གིས་བྲིས། །འདོད་ལུགས་གྲུབ་མཐའ་མང་པོ་ངས་མི་ཤེས། །བཀའ་མིན་གཏེར་མིན་ཟོབ་ཆོས་ཡིན་ཡང་ཐང༌། །སུ་མི་འཐད་ན་རང་གར་སྐྱུར་ལ་ཞོག །ཕ་གཅིག་རྟོགས་ལྡན་བླ་མའི་བཀའ་དྲིན་གྱིས། །སྣང་སྲིད་ཆོས་སྐུར་རྟོགས་པའི་གདིང་ཞིག་ཡོད། །སྟོང་གསུམ་དགྲ་རུ་ལྡང་ཡང་བག་ཚ་མེད། །འཁོར་འདས་སྒྱུ་མར་ཤེས་ནས་ཞེ་བློ་བདེ། །སྣང་བ་དཔེ་རུ་ཤར་བ་ཡིག་རིས་སུ། །འབྲི་བར་མི་འདོད་ཁ་རོག་ཉོལ་ལ་སྡོད། །ཤོག་བརྫད་སྤྲང་མོས་གང་ཤར་བྲིས་པ་དགེ། ཧ་ཧ་ཧ།། ༤༢༤༽
