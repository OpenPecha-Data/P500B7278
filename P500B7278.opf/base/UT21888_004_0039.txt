༥༣༧༽བྱིན་འབེབས། སྤྱན་འདྲེན། དངོས་གྲུབ། བསྡུ་རིམ་རྣམས་ལ། རྡོར་བསོར་བྱེད་ཚུལ་ནི། ལག་གཡས་ཀྱི་མཐེ་བོང༌། མཛུབ་མོ། གུང་མོ་གསུམ་གྱིས་རྡོ་རྗེའི་ལྟེ་བ་ཟླུམ་པོ་ནས་འཛིན་ཞིང༌། རྡོ་རྗེའི་མར་རྩེའི་འཆང་བར་མ་རེག་པ་དགོས། དེ་མིན་གཞན་རྣམས་ལ་རྡོར་བསོར་བྱེད་ཚུལ། རྡོ་རྗེ་གྱེན་བསྟན་དུ་མཐེབ་སོར་རྣམས་ཅུང་ཟད་བསྐུམས་པའི་ཚུལ་གྱིས་འཛིན་པའི་ཁྱད་པར་ཡོད་ཅེས་གསུངས་པའོ།། ༈ གསོལ་ཁའི་སྐབས། མཆོད་རྫས་རྣམས་དང་ལྷན་དུ་བྷ་ལིང་དང་སྐྱུ་གུ་སོགས་དགོས་ཏེ། སྐྱུ་གུ་ནི། ཟན་རིལ་གཅིག་ལག་གཡོན་པའི་མཐིལ་དུ་བཞག་ནས། ལག་གཡས་ཀྱི་མཛུབ་མོ་ལས། གུང་མོ་ཅུང་ཟད་མཐོ་བ། སྲིན་ལག་ཅུང་ཟད་དམའ་བ་དེ་གསུམ་ལྷན་དུ་གཤིབས་ནས་བསྐྱངས་བཞིན་ཟན་རིལ་གཅུར་བའི་དབྱིབས་འདི་འདྲ། ཟུར་བཞིའི་ཉམས་ཅན་ཡིན། རྫས་མ་ཚང་བ་རྣམས་ཀྱི་འཚབ་ཏུ་འབུལ་རྒྱུའི་ལྷག་ཆད་ཁ་སྐོང་ཆེད་ཡིན་ནོ། ཞལ་རྒྱུན་ལྟར་བྲིས་སོ།། ༥༣༨༽
