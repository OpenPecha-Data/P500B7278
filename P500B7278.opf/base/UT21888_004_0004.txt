༡༣༡༽༈ ནང་གསལ་གྱི་ཆོས་སྐུ་དེ་ལས་མ་གཡོས་བཞིན་དུ། གསལ་བའི་ཆ་དེ་ལས་ལོངས་སྐུར་བཞེངས་ནུས་པ་ཙམ་དང༌། དེ་ལས་ཕྱེད་སྤྲུལ་ལོངས་སྐུ། དེ་ལས་ཐུགས་རྗེ་སྤྲུལ་སྐུར་བཞེངས། དེ་ནས་འགྲོ་བ་གང་ལ་གང་འདུལ་གྱིས་སྟོན་པར་སྣང་བ་ཡིན། གདུལ་བྱའི་སེམས་ཅན་མ་ཟད་ཀྱི་བར་དུ་འདུལ་བྱེད་ཀྱི་སྟོན་པ་རྒྱུན་མི་ཆད་པར་འབྱུང་བ་ཡིན། སྣང་བའི་ཆོས་བདག་ལུས་ཕུང་པོ་དང་བཅས་པ་སྣང་བའི་རྒྱན། ཡེ་ཤེས་ཀྱི་རོལ་པ། མ་འགགས་པའི་རྩལ། དེ་དག་ཐམས་ཅད་ཤར་ཡང་རིག་པའི་ངང་ལས་ཤར། སྣང་ཡང་རིག་པའི་ངོ་བོར་སྣང༌། གྲོལ་ཡང་རིག་པའི་ངང་དུ་གྲོལ། རང་གི་རིག་པ་དེ་རང་སྣང་གི་སྟོན་པ་ཡིན། རང་རིག་ལས་གཞན་པའི་བླ་མ་མེད་དོ་གསུངས། བདེ་བའི་རྡོ་རྗེའི་རྨི་ལམ་མོ།། ༈ གཏེར་སྟོན་ལ། རྩ་བའི་གཟུངས་མ་མིག་གཡས་པ་ལྟ་བུ། རྩ་བའི་ཆོས་བདག་མིག་གཡོན་པ་ལྟ་བུ། སྲས་སམ། ཆོས་བདག་གམ། ཡོངས་རྫོགས་ ༡༣༢༽བཀའ་བབས་སྙིང་ལྟ་བུ། ཡི་གེའི་འདུ་བྱེད་ལག་གཡས་ལྟ་བུ། ལོངས་སྤྱོད་འདུ་བྱེད་ལག་གཡོན་ལྟ་བུ། ཆོས་བདག་ཀྭ་བ་ཞི་རྐང་གཡས་ལྟ་བུ། གདུང་མ་བརྒྱད་དམ། གྲལ་གཤམ་རྐང་གཡོན་ལྟ་བུའོ། སྟོང་པའི་ཆ་མཁའ་འགྲོའི་སྐུར་བཞེངས་པ་མཆོག་གི་གཟུངས་མ་གཅིག ནང་དོན་ཡེ་ཤེས་ལྔའི་བརྡ་རྟགས་གཙོ་བོ་ལྔ། སྟོང་ཉིད་བཅུ་དྲུག་གི་བརྡར་ཡན་ལག་གི་གཟུངས་མ་བཅུ་དྲུག འབྲས་ཆོས་ཉེར་ལྔའི་བརྡ་མཚོན་ཕྲིན་ལས་ཀྱི་གཟུངས་མ་ཉེར་ལྔའོ། ཞེས་པའང་ཤཱི་ལ་བཛྲ་གྱི་ངོར། དགོངས་འདུས་མཚོ་རྒྱལ་དཀར་ཆག་ལས་བདེ་བའི་རྡོ་རྗེས་བྲིས་པའོ།། དགེའོ།། ༡༣༡༽༈ ནང་གསལ་གྱི་ཆོས་སྐུ་དེ་ལས་མ་གཡོས་བཞིན་དུ། གསལ་བའི་ཆ་དེ་ལས་ལོངས་སྐུར་བཞེངས་ནུས་པ་ཙམ་དང༌། དེ་ལས་ཕྱེད་སྤྲུལ་ལོངས་སྐུ། དེ་ལས་ཐུགས་རྗེ་སྤྲུལ་སྐུར་བཞེངས། དེ་ནས་འགྲོ་བ་གང་ལ་གང་འདུལ་གྱིས་སྟོན་པར་སྣང་བ་ཡིན། གདུལ་བྱའི་སེམས་ཅན་མ་ཟད་ཀྱི་བར་དུ་འདུལ་བྱེད་ཀྱི་སྟོན་པ་རྒྱུན་མི་ཆད་པར་འབྱུང་བ་ཡིན།

སྣང་བའི་ཆོས་བདག་ལུས་ཕུང་པོ་དང་བཅས་པ་སྣང་བའི་རྒྱན། ཡེ་ཤེས་ཀྱི་རོལ་པ། མ་འགགས་པའི་རྩལ། དེ་དག་ཐམས་ཅད་ཤར་ཡང་རིག་པའི་ངང་ལས་ཤར། སྣང་ཡང་རིག་པའི་ངོ་བོར་སྣང༌། གྲོལ་ཡང་རིག་པའི་ངང་དུ་གྲོལ། རང་གི་རིག་པ་དེ་རང་སྣང་གི་སྟོན་པ་ཡིན། རང་རིག་ལས་གཞན་པའི་བླ་མ་མེད་དོ་གསུངས། བདེ་བའི་རྡོ་རྗེའི་རྨི་ལམ་མོ།། ༈ གཏེར་སྟོན་ལ། རྩ་བའི་གཟུངས་མ་མིག་གཡས་པ་ལྟ་བུ། རྩ་བའི་ཆོས་བདག་མིག་གཡོན་པ་ལྟ་བུ། སྲས་སམ། ཆོས་བདག་གམ། ཡོངས་རྫོགས་ ༡༣༢༽བཀའ་བབས་སྙིང་ལྟ་བུ། ཡི་གེའི་འདུ་བྱེད་ལག་གཡས་ལྟ་བུ། ལོངས་སྤྱོད་འདུ་བྱེད་ལག་གཡོན་ལྟ་བུ། ཆོས་བདག་ཀྭ་བ་ཞི་རྐང་གཡས་ལྟ་བུ། གདུང་མ་བརྒྱད་དམ། གྲལ་གཤམ་རྐང་གཡོན་ལྟ་བུའོ། སྟོང་པའི་ཆ་མཁའ་འགྲོའི་སྐུར་བཞེངས་པ་མཆོག་གི་གཟུངས་མ་གཅིག ནང་དོན་ཡེ་ཤེས་ལྔའི་བརྡ་རྟགས་གཙོ་བོ་ལྔ། སྟོང་ཉིད་བཅུ་དྲུག་གི་བརྡར་ཡན་ལག་གི་གཟུངས་མ་བཅུ་དྲུག འབྲས་ཆོས་ཉེར་ལྔའི་བརྡ་མཚོན་ཕྲིན་ལས་ཀྱི་གཟུངས་མ་ཉེར་ལྔའོ། ཞེས་པའང་ཤཱི་ལ་བཛྲ་གྱི་ངོར། དགོངས་འདུས་མཚོ་རྒྱལ་དཀར་ཆག་ལས་བདེ་བའི་རྡོ་རྗེས་བྲིས་པའོ།། དགེའོ།། ༡༣༡༽༈ ནང་གསལ་གྱི་ཆོས་སྐུ་དེ་ལས་མ་གཡོས་བཞིན་དུ། གསལ་བའི་ཆ་དེ་ལས་ལོངས་སྐུར་བཞེངས་ནུས་པ་ཙམ་དང༌། དེ་ལས་ཕྱེད་སྤྲུལ་ལོངས་སྐུ། དེ་ལས་ཐུགས་རྗེ་སྤྲུལ་སྐུར་བཞེངས། དེ་ནས་འགྲོ་བ་གང་ལ་གང་འདུལ་གྱིས་སྟོན་པར་སྣང་བ་ཡིན། གདུལ་བྱའི་སེམས་ཅན་མ་ཟད་ཀྱི་བར་དུ་འདུལ་བྱེད་ཀྱི་སྟོན་པ་རྒྱུན་མི་ཆད་པར་འབྱུང་བ་ཡིན། སྣང་བའི་ཆོས་བདག་ལུས་ཕུང་པོ་དང་བཅས་པ་སྣང་བའི་རྒྱན། ཡེ་ཤེས་ཀྱི་རོལ་པ། མ་འགགས་པའི་རྩལ། དེ་དག་ཐམས་ཅད་ཤར་ཡང་རིག་པའི་ངང་ལས་ཤར། སྣང་ཡང་རིག་པའི་ངོ་བོར་སྣང༌། གྲོལ་ཡང་རིག་པའི་ངང་དུ་གྲོལ། རང་གི་རིག་པ་དེ་རང་སྣང་གི་སྟོན་པ་ཡིན། རང་རིག་ལས་གཞན་པའི་བླ་མ་མེད་དོ་གསུངས། བདེ་བའི་རྡོ་རྗེའི་རྨི་ལམ་མོ།། ༈ གཏེར་སྟོན་ལ།

རྩ་བའི་གཟུངས་མ་མིག་གཡས་པ་ལྟ་བུ། རྩ་བའི་ཆོས་བདག་མིག་གཡོན་པ་ལྟ་བུ། སྲས་སམ། ཆོས་བདག་གམ། ཡོངས་རྫོགས་ ༡༣༢༽བཀའ་བབས་སྙིང་ལྟ་བུ། ཡི་གེའི་འདུ་བྱེད་ལག་གཡས་ལྟ་བུ། ལོངས་སྤྱོད་འདུ་བྱེད་ལག་གཡོན་ལྟ་བུ། ཆོས་བདག་ཀྭ་བ་ཞི་རྐང་གཡས་ལྟ་བུ། གདུང་མ་བརྒྱད་དམ། གྲལ་གཤམ་རྐང་གཡོན་ལྟ་བུའོ། སྟོང་པའི་ཆ་མཁའ་འགྲོའི་སྐུར་བཞེངས་པ་མཆོག་གི་གཟུངས་མ་གཅིག ནང་དོན་ཡེ་ཤེས་ལྔའི་བརྡ་རྟགས་གཙོ་བོ་ལྔ། སྟོང་ཉིད་བཅུ་དྲུག་གི་བརྡར་ཡན་ལག་གི་གཟུངས་མ་བཅུ་དྲུག འབྲས་ཆོས་ཉེར་ལྔའི་བརྡ་མཚོན་ཕྲིན་ལས་ཀྱི་གཟུངས་མ་ཉེར་ལྔའོ། ཞེས་པའང་ཤཱི་ལ་བཛྲ་གྱི་ངོར། དགོངས་འདུས་མཚོ་རྒྱལ་དཀར་ཆག་ལས་བདེ་བའི་རྡོ་རྗེས་བྲིས་པའོ།། དགེའོ།།
